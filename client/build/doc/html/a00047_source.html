229bool checkWin(char board[BOARD_SIZE][BOARD_SIZE], char player) {
230 for (int i = 0; i < BOARD_SIZE; i++) {
231 if (board[i][0] == player && board[i][1] == player && board[i][2] == player) return true;
232 if (board[0][i] == player && board[1][i] == player && board[2][i] == player) return true;
233 }
234 if (board[0][0] == player && board[1][1] == player && board[2][2] == player) return true;
235 if (board[0][2] == player && board[1][1] == player && board[2][0] == player) return true;
236 return false;
237}

281int minimax(char b[BOARD_SIZE][BOARD_SIZE], char currentPlayer, char aiPlayer, int depth) {
282 if (checkWin(b, aiPlayer)) {
283 return 10 - depth;
284 } else if (checkWin(b, opponent(aiPlayer))) {
285 return depth - 10;
286 } else if (isBoardFull(b)) {
287 return 0;
288 }
289
290 int bestScore = (currentPlayer == aiPlayer) ? -1000 : 1000;
291
292 for (int i = 0; i < BOARD_SIZE; i++) {
293 for (int j = 0; j < BOARD_SIZE; j++) {
294 if (b[i][j] != PLAYER_X && b[i][j] != PLAYER_O) {
295 b[i][j] = currentPlayer;
296 int score = minimax(b, opponent(currentPlayer), aiPlayer, depth + 1);
297 b[i][j] = '1' + (i * BOARD_SIZE + j);
298 if (currentPlayer == aiPlayer) {
299 if (score > bestScore)
300 bestScore = score;
301 } else {
302 if (score < bestScore)
303 bestScore = score;
304 }
305 }
306 }
307 }
308 return bestScore;
309}

320void bestMove(char b[BOARD_SIZE][BOARD_SIZE], char aiPlayer, int move[2]) {
321 int bestScore = -1000;
322 move[0] = -1;
323 move[1] = -1;
324 for (int i = 0; i < BOARD_SIZE; i++) {
325 for (int j = 0; j < BOARD_SIZE; j++) {
326 if (b[i][j] != PLAYER_X && b[i][j] != PLAYER_O) {
327 b[i][j] = aiPlayer;
328 int score = minimax(b, opponent(aiPlayer), aiPlayer, 0);
329 b[i][j] = '1' + (i * BOARD_SIZE + j);;
330 if (score > bestScore) {
331 bestScore = score;
332 move[0] = i;
333 move[1] = j;
334 }
335 }
336 }
